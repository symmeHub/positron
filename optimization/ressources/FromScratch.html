def F(x, a = 1, y0=0, b = 1):
    rep = y0 +np.sqrt(a*x+b**2*x**2)
    return rep

y_rep = F(x ,y0=0.1, a=0.2, b=0.1)
plt.figure()
plt.scatter(x, y)
plt.plot(x, y_rep, 'r-')
plt.show()

def fc(P, x):
    y0 = P[0]
    a = P[1]
    b = P[2]
    y_aprox = F(x, y0=y0, a=a, b=b)
    res = (y_aprox - y)**2
    res = np.sum(res)
    return np.sqrt(res)/len(x)

P = np.array([0.1, 0.2, 0.1])
fc(P, x)
